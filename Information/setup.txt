// Production setup --------------------------------------------------------------------------
1️⃣ Connect as superuser
psql -U postgres -h localhost -p 5432

2️⃣ Create the user (role)
CREATE USER immortalmplmsuser WITH PASSWORD 'immortalmplms@9090';

3️⃣ Create the database
CREATE DATABASE immortalmp_lms_db;

4️⃣ Restrict public access and allow your app user
REVOKE CONNECT ON DATABASE immortalmp_lms_db FROM PUBLIC;
GRANT CONNECT ON DATABASE immortalmp_lms_db TO immortalmplmsuser;

5️⃣ Connect to the database
\c immortalmp_lms_db

6️⃣ Create the schema
CREATE SCHEMA IF NOT EXISTS lms_api AUTHORIZATION immortalmplmsuser;

7️⃣ Harden schema permissions
REVOKE ALL ON SCHEMA lms_api FROM PUBLIC;
GRANT USAGE ON SCHEMA lms_api TO immortalmplmsuser;
GRANT CREATE ON SCHEMA lms_api TO immortalmplmsuser;

8️⃣ Set default privileges for future objects
ALTER DEFAULT PRIVILEGES FOR ROLE immortalmplmsuser IN SCHEMA lms_api
  GRANT SELECT, INSERT, UPDATE, DELETE ON TABLES TO immortalmplmsuser;

ALTER DEFAULT PRIVILEGES FOR ROLE immortalmplmsuser IN SCHEMA lms_api
  GRANT USAGE, SELECT ON SEQUENCES TO immortalmplmsuser;

9️⃣ Set user search path
ALTER ROLE immortalmplmsuser SET search_path = 'lms_api', public;

🔟 Create necessary extensions
CREATE EXTENSION IF NOT EXISTS pgcrypto;
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";


GRANT CONNECT ON DATABASE immortalmp_lms_db TO immortalmplmsuser;
GRANT USAGE, CREATE ON SCHEMA lms_api TO immortalmplmsuser;

GRANT ALL PRIVILEGES ON DATABASE immortalmp_lms_db TO immortalmplmsuser;
GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA lms_api TO immortalmplmsuser;
GRANT ALL PRIVILEGES ON ALL SEQUENCES IN SCHEMA lms_api TO immortalmplmsuser;

npx sequelize-cli db:migrate

npx sequelize-cli db:seed:all
